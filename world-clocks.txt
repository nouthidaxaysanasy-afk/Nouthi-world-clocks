function makeClock() {
  const ns = "http://www.w3.org/2000/svg";
  const svg = document.createElementNS(ns, "svg");
  svg.setAttribute("width", SIZE);
  svg.setAttribute("height", SIZE);
  svg.setAttribute("viewBox", `0 0 ${SIZE} ${SIZE}`);
  svg.classList.add("clock-face");

  const bg = document.createElementNS(ns, "circle");
  bg.setAttribute("cx", CX); bg.setAttribute("cy", CY); bg.setAttribute("r", R);
  bg.setAttribute("fill", C.face);
  bg.setAttribute("stroke", C.ring);
  bg.setAttribute("stroke-width", "1");
  svg.appendChild(bg);

  for (let i = 0; i < 60; i++) {
    const angle = (i / 60) * 2 * Math.PI - Math.PI / 2;
    const major = i % 5 === 0;
    const inner = R - (major ? 8 : 4);
    const line = document.createElementNS(ns, "line");
    line.setAttribute("x1", CX + inner * Math.cos(angle));
    line.setAttribute("y1", CY + inner * Math.sin(angle));
    line.setAttribute("x2", CX + (R - 1) * Math.cos(angle));
    line.setAttribute("y2", CY + (R - 1) * Math.sin(angle));
    line.setAttribute("stroke", C.tick);
    line.setAttribute("stroke-width", major ? "1.5" : "0.5");
    svg.appendChild(line);
  }

  function hand(len, w, color, round) {
    const el = document.createElementNS(ns, "line");
    el.setAttribute("x1", CX); el.setAttribute("y1", CY);
    el.setAttribute("x2", CX); el.setAttribute("y2", CY - len);
    el.setAttribute("stroke", color);
    el.setAttribute("stroke-width", w);
    el.setAttribute("stroke-linecap", round ? "round" : "square");
    el.setAttribute("transform", `rotate(0,${CX},${CY})`);
    svg.appendChild(el);
    return el;
  }

  const hrHand  = hand(R * 0.50, 2.5, C.hour, true);
  const minHand = hand(R * 0.75, 1.8, C.min,  true);

function setHand(hand, deg) {

function tick() {
  const now = new Date();
  const utcMs = now.getTime() + now.getTimezoneOffset() * 60000;

  allClocks.forEach(({ offset, hands, timeEl }) => {
    const local = new Date(utcMs + offset * 3600000);
    const h = local.getHours(), m = local.getMinutes(),
          s = local.getSeconds(), ms = local.getMilliseconds();

    const secDeg = (s + ms / 1000) * 6;
    const minDeg = (m + s / 60) * 6;
    const hrDeg  = ((h % 12) + m / 60) * 30;

    hands.forEach(({ hrHand, minHand, secHand }) => {
      setHand(hrHand,  hrDeg);
      setHand(minHand, minDeg);
      setHand(secHand, secDeg);
    });

    timeEl.textContent =
      String(h).padStart(2,"0") + ":" +
      String(m).padStart(2,"0") + ":" +
      String(s).padStart(2,"0");
  });

  requestAnimationFrame(tick);
}

document.querySelectorAll(".clock-card").forEach(card => {
  const city = card.querySelector(".city-name").title.toLowerCase();
  allCards.push({ card, city, group: card.closest(".timezone-group") });
});

searchInput.addEventListener("input", () => {
  const q = searchInput.value.trim().toLowerCase();
  let matches = 0;

  allCards.forEach(({ card, city, group }) => {
    card.classList.remove("highlighted");
    group.classList.remove("has-match");
  });

  if (q.length >= 2) {
    allCards.forEach(({ card, city, group }) => {
      if (city.includes(q)) {
        card.classList.add("highlighted");
        group.classList.add("has-match");
        matches++;
        setTimeout(() => card.scrollIntoView({ behavior: "smooth", block: "center" }), 50);
      }
    });
    searchHint.textContent = matches === 0 ? "Aucun résultat" : matches === 1 ? "1 ville trouvée" : `${matches} villes trouvées`;
  } else {
    searchHint.textContent = q.length === 1 ? "Saisir au moins 2 caractères" : "";
  }
});
</script>
</body>
</html>
